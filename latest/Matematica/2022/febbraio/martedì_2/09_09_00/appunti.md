$$
\lim_{x\to0^+}\frac{\ln(\sin 3x)}{\ln(\sin x)}=\frac{-\infin}{-\infin}\quad F.I.
$$


$$
\xrightarrow[\sin x \approx x]{\sin 3x \approx 3x}\lim_{x\to0^+}\frac{\ln 3x}{\ln x}
$$

$$
\xrightarrow{\ln(3)=0\ln x}
$$

$$
\lim_{x\to 0^+}\frac{\ln x}{\ln x}=1^-
$$
