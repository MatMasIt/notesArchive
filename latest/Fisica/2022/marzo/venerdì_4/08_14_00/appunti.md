$$
\Phi(\vec E) = \vec E\cdot \vec n S

$$
\Phi (\vec B) = \vec B \cdot \vec n S

$$
\Gamma(\vec E) = \vec E \cdot  \vec{\Delta s}
$$

$$
\Gamma(\vec B) = \vec B \cdot  \vec{\Delta s}
$$
